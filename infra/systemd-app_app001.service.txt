# /etc/systemd/system/app.service
[Unit]
Description=FastAPI (uvicorn) app
After=network.target

[Service]
WorkingDirectory=/opt/app
EnvironmentFile=/opt/app/app.env
ExecStart=/opt/app/.venv/bin/uvicorn main:app --host 0.0.0.0 --port 8000 --proxy-headers --forwarded-allow-ips="10.0.2.6"
Restart=always
RestartSec=2

[Install]
WantedBy=multi-user.target
